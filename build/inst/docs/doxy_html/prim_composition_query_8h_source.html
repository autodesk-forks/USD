340  struct Filter
341  {

355 
356  bool operator==(const Filter &rhs) {
357  return arcIntroducedFilter == rhs.arcIntroducedFilter &&
358  arcTypeFilter == rhs.arcTypeFilter &&
359  dependencyTypeFilter == rhs.dependencyTypeFilter &&
