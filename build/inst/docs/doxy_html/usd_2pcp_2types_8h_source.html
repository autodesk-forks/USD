139  {
140  return layer == other.layer && path == other.path;

142 
143  bool operator!=(const Pcp_SdSiteRef& other) const
144  {
145  return !(*this == other);
146  }
147 
148  bool operator<(const Pcp_SdSiteRef& other) const
149  {
150  return layer < other.layer ||

152  }
