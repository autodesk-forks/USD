47 template <typename BASE>
48 class HdPrman_Gprim : public BASE, public HdPrman_GprimBase
49 {
50 public:
51  using BaseType = BASE;
52 
53  HdPrman_Gprim(SdfPath const& id)
54  : BaseType(id)
55  {
56  }
57 
58  ~HdPrman_Gprim() override = default;
59 
60  void
61  Finalize(HdRenderParam *renderParam) override
62  {
63  HdPrman_RenderParam *param =
64  static_cast<HdPrman_RenderParam*>(renderParam);
65  const SdfPath& id = BASE::GetId();
66  riley::Riley *riley = param->AcquireRiley();
67 
68  // Release retained conversions of coordSys bindings.
69  param->ReleaseCoordSysBindings(id);
70 
71  // Delete instances before deleting the prototypes they use.
72  for (const auto &instId: _instanceIds) {
73  if (instId != riley::GeometryInstanceId::InvalidId()) {
74  riley->DeleteGeometryInstance(
75  riley::GeometryPrototypeId::InvalidId(), instId);
76  }
77  }
78  _instanceIds.clear();
79 
80  // delete instances owned by the instancer.
81  if (HdPrmanInstancer* instancer = param->GetInstancer(
82  BASE::GetInstancerId())) {
83  instancer->Depopulate(renderParam, id);
84  }
85 
86  for (const auto &protoId: _prototypeIds) {
87  if (protoId != riley::GeometryPrototypeId::InvalidId()) {
88  riley->DeleteGeometryPrototype(protoId);
89  }
90  }
91  _prototypeIds.clear();
92  }
93 
94  void Sync(HdSceneDelegate* sceneDelegate,
95  HdRenderParam* renderParam,
96  HdDirtyBits* dirtyBits,
97  TfToken const &reprToken) override;
98

100  HdDirtyBits GetInitialDirtyBitsMask() const override = 0;
101 
102  HdDirtyBits
103  _PropagateDirtyBits(HdDirtyBits bits) const override

105  // By default, just return the same dirty bits we recieved.

111  HdDirtyBits *dirtyBits) override

116  }

123  return false;

148 template <typename BASE>
149 void
150 HdPrman_Gprim<BASE>::Sync(HdSceneDelegate* sceneDelegate,
151  HdRenderParam* renderParam,
152  HdDirtyBits* dirtyBits,
153  TfToken const &reprToken)
154 {
155  HD_TRACE_FUNCTION();
156  HF_MALLOC_TAG_FUNCTION();
157  TF_UNUSED(reprToken);
158 
159  HdPrman_RenderParam *param =
160  static_cast<HdPrman_RenderParam*>(renderParam);
161 
162  // Riley API.
163  riley::Riley *riley = param->AcquireRiley();
164 
165  // Update instance bindings.
166  BASE::_UpdateInstancer(sceneDelegate, dirtyBits);
167 
168  // Prim id
169  SdfPath const& id = BASE::GetId();
170  SdfPath const& instancerId = BASE::GetInstancerId();
171  const bool isHdInstance = !instancerId.IsEmpty();
172  SdfPath primPath = sceneDelegate->GetScenePrimPath(id, 0, nullptr);
173 
174  // Prman has a default value for identifier:id of 0 (in case of ray miss),
175  // while Hydra treats id -1 as the clear value. We map Prman primId as
176  // (Hydra primId + 1) to get around this, here and in
177  // hdPrman/framebuffer.cpp.
178  const int32_t primId = BASE::GetPrimId() + 1;
179 
180  // Sample transform
181  HdTimeSampleArray<GfMatrix4d, HDPRMAN_MAX_TIME_SAMPLES> xf;
182  sceneDelegate->SampleTransform(id, &xf);
183 
184  // If blur is explicitly disabled, use single time 0 sample
185  const bool enableMotionBlur =
186  param->IsMotionBlurEnabled() &&
187  HdPrman_IsMotionBlurPrimvarEnabled(sceneDelegate, id) &&
188  HdPrman_GetNumXformSamples(sceneDelegate, id) >= 2;
189 
190  if (!enableMotionBlur) {
191  xf.values[0] = xf.Resample(0.f);
192  xf.times[0] = 0.f;
193  xf.Resize(1);
194  }
195 
196  // Update visibility so thet rprim->IsVisible() will work in render pass
197  if (HdChangeTracker::IsVisibilityDirty(*dirtyBits, id)) {
198  BASE::_UpdateVisibility(sceneDelegate, dirtyBits);
199  }
200 
201  // Resolve material binding. Default to fallbackGprimMaterial.
202  if (*dirtyBits & HdChangeTracker::DirtyMaterialId) {
203 #if HD_API_VERSION < 37
204  BASE::_SetMaterialId(sceneDelegate->GetRenderIndex().GetChangeTracker(),
205  sceneDelegate->GetMaterialId(id));
206 #else
207  BASE::SetMaterialId(sceneDelegate->GetMaterialId(id));
208 #endif
209  }
210  riley::MaterialId materialId = _GetFallbackMaterial(param);
211  riley::DisplacementId dispId = riley::DisplacementId::InvalidId();
212  const SdfPath & hdMaterialId = BASE::GetMaterialId();
213  HdPrman_ResolveMaterial(sceneDelegate, hdMaterialId, riley, &materialId, &dispId);
214 
215  // Convert (and cache) coordinate systems.
216  riley::CoordinateSystemList coordSysList = {0, nullptr};
217  if (HdPrman_RenderParam::RileyCoordSysIdVecRefPtr convertedCoordSys =
218  param->ConvertAndRetainCoordSysBindings(sceneDelegate, id)) {
219  coordSysList.count = convertedCoordSys->size();
220  coordSysList.ids = convertedCoordSys->data();
221  }
222 
223  // Hydra dirty bits corresponding to PRMan prototype attributes (also called
224  // "primitive variables" but not synonymous with USD primvars). See prman
225  // docs at https://rmanwiki.pixar.com/display/REN24/Primitive+Variables.
226  static const int prmanProtoAttrBits =
227  HdChangeTracker::DirtyPoints |
228  HdChangeTracker::DirtyNormals |
229  HdChangeTracker::DirtyWidths |
230  HdChangeTracker::DirtyTopology;
231 
232  // Hydra dirty bits corresponding to prman instance attributes. See prman
233  // docs at https://rmanwiki.pixar.com/display/REN24/Instance+Attributes.
234  static const int prmanInstAttrBits =
235  HdChangeTracker::DirtyMaterialId |
236  HdChangeTracker::DirtyTransform |
237  HdChangeTracker::DirtyVisibility |
238  HdChangeTracker::DirtyDoubleSided |
239  HdChangeTracker::DirtySubdivTags |
240  HdChangeTracker::DirtyVolumeField |
241  HdChangeTracker::DirtyCategories |
242  HdChangeTracker::DirtyPrimvar;
243 
244  //
245  // Create or modify Riley geometry prototype(s).
246  //
247  std::vector<riley::MaterialId> subsetMaterialIds;
248  std::vector<SdfPath> subsetPaths;
249  {
250  RtUString primType;
251  HdGeomSubsets geomSubsets;
252  RtPrimVarList primvars = _ConvertGeometry(param, sceneDelegate, id,
253  &primType, &geomSubsets);
254 
255  // Transfer material opinions of primvars.
256  HdPrman_TransferMaterialPrimvarOpinions(sceneDelegate, hdMaterialId,
257  primvars);
258 
259  // Adjust _prototypeIds array.
260  const size_t oldCount = _prototypeIds.size();
261  const size_t newCount = std::max((size_t) 1, geomSubsets.size());
262  if (newCount != oldCount) {
263  for (const auto &oldPrototypeId: _prototypeIds) {
264  if (oldPrototypeId != riley::GeometryPrototypeId::InvalidId()) {
265  riley->DeleteGeometryPrototype(oldPrototypeId);
266  }
267  }
268  _prototypeIds.resize(newCount,
269  riley::GeometryPrototypeId::InvalidId());
270  }
271 
272  // Update Riley geom prototypes.
273  if (geomSubsets.empty()) {
274  // Common case: no subsets.
275  TF_VERIFY(newCount == 1);
276  TF_VERIFY(_prototypeIds.size() == 1);
277  primvars.SetString(RixStr.k_stats_prototypeIdentifier,
278  RtUString(primPath.GetText()));
279  if (_prototypeIds[0] == riley::GeometryPrototypeId::InvalidId()) {
280  TRACE_SCOPE("riley::CreateGeometryPrototype");
281  _prototypeIds[0] = riley->CreateGeometryPrototype(
282  riley::UserId(
283  stats::AddDataLocation(primPath.GetText()).GetValue()),
284  primType, dispId, primvars);
285  } else if (*dirtyBits & prmanProtoAttrBits) {
286  TRACE_SCOPE("riley::ModifyGeometryPrototype");
287  riley->ModifyGeometryPrototype(primType, _prototypeIds[0],
288  &dispId, &primvars);
289  }
290  } else {
291  // Subsets case.
292  // We resolve materials here, and hold them in subsetMaterialIds:
293  // Displacement networks are passed to the geom prototype;
294  // material networks are passed to the instances.
295  subsetMaterialIds.reserve(geomSubsets.size());
296 
297  // We also cache the subset paths for re-use when creating the instances
298  subsetPaths.reserve(geomSubsets.size());
299 
300  for (size_t j=0; j < geomSubsets.size(); ++j) {
301  auto& prototypeId = _prototypeIds[j];
302  HdGeomSubset &subset = geomSubsets[j];
303 
304  // Convert indices to int32_t and set as k_shade_faceset.
305  std::vector<int32_t> int32Indices(subset.indices.begin(),
306  subset.indices.end());
307  primvars.SetIntegerArray(RixStr.k_shade_faceset,
308  int32Indices.data(),
309  int32Indices.size());
310  // Look up material override for the subset (if any)
311  riley::MaterialId subsetMaterialId = materialId;
312  riley::DisplacementId subsetDispId = dispId;
313  if (subset.materialId.IsEmpty()) {
314  subset.materialId = hdMaterialId;
315  }
316  HdPrman_ResolveMaterial(sceneDelegate, subset.materialId,
317  riley, &subsetMaterialId, &subsetDispId);
318  subsetMaterialIds.push_back(subsetMaterialId);
319 
320  // Look up the path for the subset
321  SdfPath subsetPath = sceneDelegate->GetScenePrimPath(subset.id, 0, nullptr);
322  subsetPaths.push_back(subsetPath);
323  primvars.SetString(RixStr.k_stats_prototypeIdentifier,
324  RtUString(subsetPath.GetText()));
325 
326  if (prototypeId == riley::GeometryPrototypeId::InvalidId()) {
327  TRACE_SCOPE("riley::CreateGeometryPrototype");
328  prototypeId =
329  riley->CreateGeometryPrototype(
330  riley::UserId(
331  stats::AddDataLocation(subsetPath.GetText()).GetValue()),
332  primType, dispId, primvars);
333  } else if (*dirtyBits & prmanProtoAttrBits) {
334  TRACE_SCOPE("riley::ModifyGeometryPrototype");
335  riley->ModifyGeometryPrototype(primType, prototypeId,
336  &subsetDispId, &primvars);
337  }
338  }
339  }
340  *dirtyBits &= ~prmanProtoAttrBits;
341  }
342 
343  //
344  // Stop here, or also create geometry instances?
345  //
346  if (_PrototypeOnly()) {
347  *dirtyBits &= ~HdChangeTracker::AllSceneDirtyBits;
348  return;
349  }
350 
351  //
352  // Create or modify Riley geometry instances.
353  //
354 
355  // Resolve attributes.
356  RtParamList attrs = param->ConvertAttributes(sceneDelegate, id, true);
357 
358  // Add "identifier:id" with the prim id.
359  attrs.SetInteger(RixStr.k_identifier_id, primId);
360 
361  if (!isHdInstance) {
362  // Simple case: Singleton instance.
363  // Convert transform.
364  TfSmallVector<RtMatrix4x4, HDPRMAN_MAX_TIME_SAMPLES> xf_rt(xf.count);
365  for (size_t i=0; i < xf.count; ++i) {
366  xf_rt[i] = HdPrman_Utils::GfMatrixToRtMatrix(xf.values[i]);
367  }
368  const riley::Transform xform = {
369  unsigned(xf.count),
370  xf_rt.data(),
371  xf.times.data()};
372 
373  // Add "identifier:id2" with the instance number.
374  // XXX Do we want to distinguish facesets here?
375  attrs.SetInteger(RixStr.k_identifier_id2, 0);
376 
377  // Adjust _instanceIds array.
378  const size_t oldCount = _instanceIds.size();
379  const size_t newCount = _prototypeIds.size();
380  if (newCount != oldCount) {
381  for (const auto &oldInstanceId: _instanceIds) {
382  if (oldInstanceId != riley::GeometryInstanceId::InvalidId()) {
383  riley->DeleteGeometryInstance(
384  riley::GeometryPrototypeId::InvalidId(), oldInstanceId);
385  }
386  }
387  _instanceIds.resize(
388  newCount,
389  riley::GeometryInstanceId::InvalidId());
390  }
391 
392  // Create or modify Riley instances corresponding to a
393  // singleton Hydra instance.
394  TF_VERIFY(_instanceIds.size() == _prototypeIds.size());
395  for (size_t j=0; j < _prototypeIds.size(); ++j) {
396  auto const& prototypeId = _prototypeIds[j];
397  auto& instanceId = _instanceIds[j];
398  auto instanceMaterialId = materialId;
399  RtParamList finalAttrs = attrs; // copy
400 
401  // If a subset path was cached, use it. If not, use the prim path.
402  SdfPath* subsetPath(&primPath);
403  if (!subsetPaths.empty()) {
404  subsetPath = &subsetPaths[j];
405  }
406 
407  finalAttrs.SetString(RixStr.k_identifier_name,
408  RtUString(subsetPath->GetText()));
409 
410  // If a valid subset material was bound, use it.
411  if (!subsetMaterialIds.empty()) {
412  TF_VERIFY(j < subsetMaterialIds.size());
413  instanceMaterialId = subsetMaterialIds[j];
414  }
415 
416  if (instanceId == riley::GeometryInstanceId::InvalidId()) {
417  TRACE_SCOPE("riley::CreateGeometryInstance");
418  instanceId = riley->CreateGeometryInstance(
419  riley::UserId(
420  stats::AddDataLocation(subsetPath->GetText()).GetValue()),
421  riley::GeometryPrototypeId::InvalidId(), prototypeId,
422  instanceMaterialId, coordSysList, xform, finalAttrs);
423  } else if (*dirtyBits & prmanInstAttrBits) {
424  TRACE_SCOPE("riley::ModifyGeometryInstance");
425  riley->ModifyGeometryInstance(
426  riley::GeometryPrototypeId::InvalidId(),
427  instanceId, &instanceMaterialId, &coordSysList, &xform,
428  &finalAttrs);
429  }
430  }
431  *dirtyBits &= ~prmanInstAttrBits;
432  } else if ((*dirtyBits & prmanInstAttrBits)
433  || HdChangeTracker::IsInstancerDirty(*dirtyBits, instancerId)) {
434  // This gprim is a prototype of a hydra instancer. (It is not itself an
435  // instancer because it is a gprim.) The riley geometry prototypes have
436  // already been synced above, and those are owned by this gprim instance.
437  // We need to tell the hdprman instancer to sync its riley instances for
438  // these riley prototypes.
439  //
440  // We won't make any riley instances here. The hdprman instancer will
441  // own the riley instances instead.
442  //
443  // We only need to do this if dirtyBits says the instancer is dirty.
444 
445  HdRenderIndex &renderIndex = sceneDelegate->GetRenderIndex();
446 
447  // first, sync the hydra instancer and its parents, from the bottom up.
448  // (note: this is transitional code, it should be done by the render index...)
449  HdInstancer::_SyncInstancerAndParents(renderIndex, instancerId);
450 
451  if (subsetMaterialIds.size() == 0) {
452  subsetMaterialIds.push_back(materialId);
453  }
454  if (subsetPaths.size() == 0) {
455  subsetPaths.push_back(primPath);
456  }
457  TF_VERIFY(_prototypeIds.size() == subsetMaterialIds.size() &&
458  _prototypeIds.size() == subsetPaths.size(),
459  "size mismatch (%lu, %lu, %lu)\n", _prototypeIds.size(),
460  subsetMaterialIds.size(), subsetPaths.size());
461 
462  // next, tell the hdprman instancer to sync the riley instances
463  HdPrmanInstancer *instancer = static_cast<HdPrmanInstancer*>(
464  renderIndex.GetInstancer(instancerId));
465  if (instancer) {
466  instancer->Populate(
467  renderParam,
468  dirtyBits,
469  id,
470  _prototypeIds,
471  coordSysList,
472  attrs, xf,
473  subsetMaterialIds,
474  subsetPaths);
475  }
476  }
477  *dirtyBits &= ~HdChangeTracker::AllSceneDirtyBits;
478 }
