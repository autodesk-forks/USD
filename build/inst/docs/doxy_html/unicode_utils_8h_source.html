108 class TfUtf8CodePointIterator final {
109 public:
110  using iterator_category = std::forward_iterator_tag;
111  using value_type = uint32_t;
112  using difference_type = std::ptrdiff_t;
113  using pointer = void;
114  using reference = uint32_t;
115 
116  static constexpr uint32_t INVALID_CODE_POINT = 0xFFFD;
117 
125  uint32_t operator* () const
126  {
127  // If the current UTF-8 character is invalid, instead of
128  // throwing an exception, _GetCodePoint signals this is
129  // bad by setting the code point to 0xFFFD (this mostly happens
130  // when a high / low private surrogate is used)
131  // TODO: note that this isn't precisely conformant, as we
132  // likely consumed an entire sequence when a subset would have
133  // been invalid e.g. the byte sequence C2 41 would be detected
134  // as a 2-byte sequence, but it is invalid because of the
135  // second byte signature by the standard, this should process
136  // the C2 as invalid but consume 41 as a valid 1-byte UTF-8
137  // character
138  return _GetCodePoint();
139  }
140

143  {
144  return this->_it;
145  }
146

152  return (this->_it == rhs._it);
153  }
154 
158  bool operator!= (const TfUtf8CodePointIterator& rhs) const

160  return (this->_it != rhs._it);

162 
168  TfUtf8CodePointIterator& operator++ ()
169  {
170  // note that in cases where the encoding is invalid, we move to the
171  // next byte this is necessary because otherwise the iterator would
172  // never advanced and the end condition of == iterator::end() would
173  // never be satisfied
174  _EncodingLength encodingLength = _GetEncodingLength();
175  std::advance(_it, (encodingLength != 0) ? encodingLength : 1);
176  if (_IsPastTheEnd()) {
177  _it = _end;
178  }
179  return *this;
180  }
181 
185  TfUtf8CodePointIterator operator++ (int)
186  {
187  // note that in cases where the encoding is invalid, we move to the
188  // next byte this is necessary because otherwise the iterator would
189  // never advanced and the end condition of == iterator::end() would
190  // never be satisfied
191  auto temp = *this;
192  ++(*this);
193  return temp;
194  }

221 private:
222  // Constructs an iterator that can read UTF-8 character sequences from
223  // the given starting string_view iterator \a it. \a end is used as a
224  // guard against reading byte sequences past the end of the source string.
225  TfUtf8CodePointIterator(
226  const std::string_view::const_iterator& it,
227  const std::string_view::const_iterator& end) : _it(it), _end(end) {
228  TF_DEV_AXIOM(_it <= _end);
229  }
230 
231  using _EncodingLength = unsigned char;
232 
233  // Retrieves the variable encoding length of the UTF-8 character
234  // currently pointed to by the iterator. This can be 1, 2, 3, or 4
235  // depending on the encoding of the UTF-8 character. If the encoding
236  // cannot be determined, this method will return 0.
237  _EncodingLength _GetEncodingLength() const
238  {
239  // already at the end, no valid character sequence
240  if (_IsPastTheEnd())
241  {
242  return 0;
243  }
244  // determine what encoding length the character is
245  // 1-byte characters have a leading 0 sequence
246  // 2-byte characters have a leading 110 sequence
247  // 3-byte characters have a leading 1110 sequence
248  // 4-byte characters have a leading 11110 sequence
249  unsigned char x = static_cast<unsigned char>(*_it);
250  if (x < 0x80)
251  {
252  return 1;
253  }
254  else if ((x >> 5) == 0x6)
255  {
256  return 2;
257  }
258  else if ((x >> 4) == 0xe)
259  {
260  return 3;
261  }
262  else if ((x >> 3) == 0x1e)
263  {
264  return 4;
265  }
266  else
267  {
268  // can't determine encoding, this is an error
269  return 0;
270  }
271  }
272 
273  // Retrieves the Unicode code point of the next character in the UTF-8
274  // encoded sequence (defined by \a begin) and returns the value in
275  // \a codePoint. This method will return \a true if the encoded
276  // sequence is valid. If the encoding is invalid, this method will
277  // return \a false and \a codePoint will be set to 0.
278  TF_API uint32_t _GetCodePoint() const;
279 
280  // Returns true if the iterator at or past the end and can no longer be
281  // dereferenced.
282  bool _IsPastTheEnd() const
283  {
284  return _it >= _end;
285  }
286 
287  std::string_view::const_iterator _it;
288  std::string_view::const_iterator _end;
289 
290  friend class TfUtf8CodePointView;
291 };
