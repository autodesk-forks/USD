46 // Helper for swap-based asynchronous destruction that synthesizes move
47 // construction and assignment using swap.
48 template <class T>
49 struct Work_AsyncSwapDestroyHelper {
50  Work_AsyncSwapDestroyHelper() = default;
51 
52  Work_AsyncSwapDestroyHelper(Work_AsyncSwapDestroyHelper const&) = delete;
53  Work_AsyncSwapDestroyHelper& operator=(
54  Work_AsyncSwapDestroyHelper const&) = delete;
55 
56  Work_AsyncSwapDestroyHelper(Work_AsyncSwapDestroyHelper &&other)
57  : obj()
58  {
59  using std::swap;
60  swap(obj, other.obj);
61  }
62 
63  Work_AsyncSwapDestroyHelper& operator=(
64  Work_AsyncSwapDestroyHelper &&other)
65  {
66  using std::swap;

69  }
70 
71  void operator()() const { /* do nothing */ }

73 };

82 template <class T>
83 void WorkSwapDestroyAsync(T &obj)
84 {
