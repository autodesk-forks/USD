303 class GlfAnyGLContextScopeHolder final : private GlfGLContextScopeHolder
304 {
305 public:
306  GlfAnyGLContextScopeHolder() :
307  GlfGLContextScopeHolder(_GetAnyContext())
308  {
309  // Do nothing
310  }
311 
312 private:
313  static GlfGLContextSharedPtr _GetAnyContext()
314  {
315  if (GlfGLContext::IsInitialized() && ArchIsMainThread()) {
316  GlfGLContextSharedPtr const current =
317  GlfGLContext::GetCurrentGLContext();
318  if (!(current &&
319  current->IsValid() &&
320  current->IsSharing(GlfGLContext::GetSharedGLContext()))) {
321  return GlfGLContext::GetSharedGLContext();
322  }
323  }
324  return GlfGLContextSharedPtr();
325  }
326 };
