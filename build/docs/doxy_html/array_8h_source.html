73 // Private base class helper for VtArray implementation.

75 {
76 public:
77  Vt_ArrayBase() : _shapeData { 0 }, _foreignSource(nullptr) {}

79  Vt_ArrayBase(Vt_ArrayForeignDataSource *foreignSrc)
80  : _shapeData { 0 }, _foreignSource(foreignSrc) {}

84  other._shapeData.clear();

88  Vt_ArrayBase &operator=(Vt_ArrayBase const &other) = default;
89  Vt_ArrayBase &operator=(Vt_ArrayBase &&other) {
90  if (this == &other)
91  return *this;
92  *this = other;
93  other._shapeData.clear();
94  other._foreignSource = nullptr;

98 protected:
99  // Control block header for native data representation. Houses refcount and
100  // capacity. For arrays with native data, this structure always lives
101  // immediately preceding the start of the array's _data in memory. See
102  // _GetControlBlock() for details.
103  struct _ControlBlock {
104  _ControlBlock() : nativeRefCount(0), capacity(0) {}
105  _ControlBlock(size_t initCount, size_t initCap)
106  : nativeRefCount(initCount), capacity(initCap) {}
107  mutable std::atomic<size_t> nativeRefCount;
108  size_t capacity;
109  };
110 
111  _ControlBlock &_GetControlBlock(void *nativeData) {
112  TF_DEV_AXIOM(!_foreignSource);
113  return *(reinterpret_cast<_ControlBlock *>(nativeData) - 1);
114  }
115 
116  _ControlBlock const &_GetControlBlock(void *nativeData) const {
117  TF_DEV_AXIOM(!_foreignSource);
118  return *(reinterpret_cast<_ControlBlock *>(nativeData) - 1);
119  }
120 
121  // Mutable ref count, as is standard.
122  std::atomic<size_t> &_GetNativeRefCount(void *nativeData) const {
123  return _GetControlBlock(nativeData).nativeRefCount;
124  }
125 
126  size_t &_GetCapacity(void *nativeData) {
127  return _GetControlBlock(nativeData).capacity;
128  }
129  size_t const &_GetCapacity(void *nativeData) const {
130  return _GetControlBlock(nativeData).capacity;
131  }

133  VT_API void _DetachCopyHook(char const *funcName) const;

135  Vt_ShapeData _shapeData;
136  Vt_ArrayForeignDataSource *_foreignSource;
137 };

428  template <typename... Args>
429  void emplace_back(Args&&... args) {
430  // If this is a non-pxr array with rank > 1, disallow push_back.
431  if (ARCH_UNLIKELY(_shapeData.otherDims[0])) {
432  TF_CODING_ERROR("Array rank %u != 1", _shapeData.GetRank());
433  return;
434  }
435  // If we don't own the data, or if we need more space, realloc.
436  size_t curSize = size();
437  if (ARCH_UNLIKELY(
438  _foreignSource || !_IsUnique() || curSize == capacity())) {
439  value_type *newData = _AllocateCopy(
440  _data, _CapacityForSize(curSize + 1), curSize);
441  ::new (static_cast<void*>(newData + curSize)) value_type(
442  std::forward<Args>(args)...);
443  _DecRef();
444  _data = newData;
445  }
446  else {
447  ::new (static_cast<void*>(_data + curSize)) value_type(
448  std::forward<Args>(args)...);
449  }
450  // Adjust size.
451  ++_shapeData.totalSize;
452  }

586  template <class FillElemsFn>
587  void resize(size_t newSize, FillElemsFn &&fillElems) {
588  const size_t oldSize = size();
589  if (oldSize == newSize) {
590  return;
591  }
592  if (newSize == 0) {
593  clear();
594  return;
595  }
596 
597  const bool growing = newSize > oldSize;
598  value_type *newData = _data;
599 
600  if (!_data) {
601  // Allocate newSize elements and initialize.
602  newData = _AllocateNew(newSize);
603  std::forward<FillElemsFn>(fillElems)(newData, newData + newSize);
604  }
605  else if (_IsUnique()) {
606  if (growing) {
607  if (newSize > _GetCapacity(_data)) {
608  newData = _AllocateCopy(_data, newSize, oldSize);
609  }
610  // fill with newly added elements from oldSize to newSize.
611  std::forward<FillElemsFn>(fillElems)(newData + oldSize,
612  newData + newSize);
613  }
614  else {
615  // destroy removed elements
616  for (auto *cur = newData + newSize,
617  *end = newData + oldSize; cur != end; ++cur) {
618  cur->~value_type();
619  }
620  }
621  }
622  else {
623  newData =
624  _AllocateCopy(_data, newSize, growing ? oldSize : newSize);
625  if (growing) {
626  // fill with newly added elements from oldSize to newSize.
627  std::forward<FillElemsFn>(fillElems)(newData + oldSize,
628  newData + newSize);
629  }
630  }
631 
632  // If we created new data, clean up the old and move over to the new.
633  if (newData != _data) {
634  _DecRef();
635  _data = newData;
636  }
637  // Adjust size.
638  _shapeData.totalSize = newSize;
639  }
640 
642  void clear() {
643  if (!_data)
644  return;
645  if (_IsUnique()) {
646  // Clear out elements, run dtors, keep capacity.
647  for (value_type *p = _data, *e = _data + size(); p != e; ++p) {
648  p->~value_type();
649  }
650  }
651  else {
652  // Detach to empty.
653  _DecRef();
654  }
655  _shapeData.totalSize = 0;
656  }

690  iterator erase(const_iterator first, const_iterator last) {
691  if (first == last){
692  return std::next(begin(), std::distance(cbegin(), last));
693  }
694  if ((first == cbegin()) && (last == cend())){
695  clear();
696  return end();
697  }
698  // Given the previous two conditions, we know that we are removing
699  // at least one element and the result array will contain at least one
700  // element.
701  value_type* removeStart = std::next(_data, std::distance(cbegin(), first));
702  value_type* removeEnd = std::next(_data, std::distance(cbegin(), last));
703  value_type* endIt = std::next(_data, size());
704  size_t newSize = size() - std::distance(first, last);
705  if (_IsUnique()){
706  // If the array is unique, we can simply move the tail elements
707  // and free to the end of the array.
708  value_type* deleteIt = std::move(removeEnd, endIt, removeStart);
709  for (; deleteIt != endIt; ++deleteIt) {
710  deleteIt->~value_type();
711  }
712  _shapeData.totalSize = newSize;
713  return iterator(removeStart);
714  } else{
715  // If the array is not unique, we want to avoid copying the
716  // elements in the range we are erasing. We allocate a
717  // new buffer and copy the head and tail ranges, omitting
718  // [first, last)
719  value_type* newData = _AllocateNew(newSize);
720  value_type* newMiddle = std::uninitialized_copy(
721  _data, removeStart, newData);
722  value_type* newEnd = std::uninitialized_copy(
723  removeEnd, endIt, newMiddle);
724  TF_DEV_AXIOM(newEnd == std::next(newData, newSize));
725  TF_DEV_AXIOM(std::distance(newData, newMiddle) ==
726  std::distance(_data, removeStart));
727  _DecRef();
728  _data = newData;
729  _shapeData.totalSize = newSize;
730  return iterator(newMiddle);
731  }

867  inline size_t _CapacityForSize(size_t sz) const {
868  // Currently just successive powers of two.

876  value_type *_AllocateNew(size_t capacity) {
877  TfAutoMallocTag2 tag("VtArray::_AllocateNew", __ARCH_PRETTY_FUNCTION__);
878  // Need space for the control block and capacity elements.
879  // Exceptionally large capacity requests can overflow the arithmetic
880  // here. If that happens we'll just attempt to allocate the max size_t
881  // value and let new() throw.
882  size_t numBytes = (capacity <= max_size())
883  ? sizeof(_ControlBlock) + capacity * sizeof(value_type)
884  : std::numeric_limits<size_t>::max();
885  void *data = ::operator new(numBytes);
886  // Placement-new a control block.
887  ::new (data) _ControlBlock(/*count=*/1, capacity);
888  // Data starts after the block.
889  return reinterpret_cast<value_type *>(
890  static_cast<_ControlBlock *>(data) + 1);
891  }

893  value_type *_AllocateCopy(value_type *src, size_t newCapacity,
894  size_t numToCopy) {
895  // Allocate and copy elements.
896  value_type *newData = _AllocateNew(newCapacity);
897  std::uninitialized_copy(src, src + numToCopy, newData);
898  return newData;
899  }

901  void _DecRef() {
902  if (!_data)
903  return;
904  if (ARCH_LIKELY(!_foreignSource)) {
905  // Drop the refcount. If we take it to zero, destroy the data.
906  if (_GetNativeRefCount(_data).fetch_sub(
907  1, std::memory_order_release) == 1) {
908  std::atomic_thread_fence(std::memory_order_acquire);
909  for (value_type *p = _data, *e = _data + _shapeData.totalSize;
910  p != e; ++p) {
911  p->~value_type();
912  }
913  ::operator delete(static_cast<void *>(
914  std::addressof(_GetControlBlock(_data))));
915  }
916  }
917  else {
918  // Drop the refcount in the foreign source. If we take it to zero,
919  // invoke the function pointer to alert the foreign source.
920  if (_foreignSource->_refCount.fetch_sub(
921  1, std::memory_order_release) == 1) {
922  std::atomic_thread_fence(std::memory_order_acquire);
923  _foreignSource->_ArraysDetached();
924  }
925  }
926  _foreignSource = nullptr;
927  _data = nullptr;
928  }

1000  if (!lhs.empty() && !rhs.empty() && lhs.size() != rhs.size()) { \

1002  return VtArray<T>(); \
1003  } \
1004  /* promote empty vecs to vecs of zeros */ \
1005  const bool leftEmpty = lhs.size() == 0, rightEmpty = rhs.size() == 0; \

1007  T zero = VtZero<T>(); \
